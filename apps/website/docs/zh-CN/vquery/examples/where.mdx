# Where

## basic

Basic Where Example

```tsx preview
import { VQueryResultRender } from '@components'

export default () => {
  const vqueryConfig = {
    datasetId: 'where-basic',
    description: 'Basic Where Example',
    vquery: {
      select: ['name', 'department', 'salary'],
      where: {
        op: 'and',
        conditions: [
          {
            field: 'department',
            op: '=',
            value: 'Engineering',
          },
        ],
      },
    },
    dataset: [
      {
        id: 1,
        name: 'Alice',
        age: 25,
        department: 'HR',
        salary: 5000,
      },
      {
        id: 2,
        name: 'Bob',
        age: 30,
        department: 'Engineering',
        salary: 8000,
      },
      {
        id: 3,
        name: 'Charlie',
        age: 35,
        department: 'Engineering',
        salary: 9000,
      },
      {
        id: 4,
        name: 'David',
        age: 40,
        department: 'HR',
        salary: 6000,
      },
      {
        id: 5,
        name: 'Eve',
        age: 22,
        department: 'Marketing',
        salary: 4000,
      },
    ],
    schema: [
      {
        name: 'id',
        type: 'number',
      },
      {
        name: 'name',
        type: 'string',
      },
      {
        name: 'age',
        type: 'number',
      },
      {
        name: 'department',
        type: 'string',
      },
      {
        name: 'salary',
        type: 'number',
      },
    ],
  }

  return <VQueryResultRender vqueryConfig={vqueryConfig} />
}
```

## between_operator

Where with BETWEEN / NOT BETWEEN Operator

```tsx preview
import { VQueryResultRender } from '@components'

export default () => {
  const vqueryConfig = {
    datasetId: 'where-between-operator',
    description: 'Where with BETWEEN / NOT BETWEEN Operator',

      select: ['id', 'name', 'age', 'gender'],
      where: {
        op: 'or',
        conditions: [
          {
            op: 'and',
            conditions: [
              {
                field: 'age',
                op: 'between',
                value: [18, 30],
              },
              {
                field: 'gender',
                op: 'in',
```
