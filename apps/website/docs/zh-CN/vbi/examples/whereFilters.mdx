# whereFilters

## add-multiple-where-filters

添加多个Where过滤条件（链式调用）

```tsx preview
import { VBI } from '@visactor/vbi'
import { VSeedRender } from '@components'
import { useEffect, useState } from 'react'

export default () => {
  const [vseed, setVSeed] = useState(null)

  useEffect(() => {
    const run = async () => {
      const schema = [
        { name: 'province', type: 'string' },
        { name: 'product_type', type: 'string' },
        { name: 'sales', type: 'number' },
      ]
      const dataset = [
        { province: 'province0', product_type: 'product_type0', sales: 100 },
        { province: 'province1', product_type: 'product_type1', sales: 200 },
        { province: 'province2', product_type: 'product_type2', sales: 300 },
        { province: 'province3', product_type: 'product_type3', sales: 400 },
        { province: 'province4', product_type: 'product_type4', sales: 500 },
      ]

      VBI.registerConnector('demo', async () => ({
        discoverSchema: async () => schema,
        query: async () => ({ dataset }),
      }))

      const builder = VBI.from({
        connectorId: 'demo',
        chartType: 'bar',
        dimensions: [{ field: 'province', alias: '省份' }],
        measures: [{ field: 'sales', alias: '销售额', encoding: 'yAxis', aggregate: { func: 'sum' } }],
        whereFilters: [],
        havingFilters: [],
        theme: 'light',
        locale: 'zh-CN',
        version: 1,
        limit: 20,
      })

      const applyBuilder = (builder: VBIBuilder) => {
        builder.whereFilters
          .add('product_type', (n) => n.setOperator('eq').setValue('办公用品'))
          .add('province', (n) => n.setOperator('in').setValue(['浙江', '江苏']))
      }
      applyBuilder(builder)

      const result = await builder.buildVSeed()
      setVSeed(result)
    }
    run()
  }, [])

  if (!vseed) return <div>Loading...</div>

  return <VSeedRender vseed={vseed} />
}
```
