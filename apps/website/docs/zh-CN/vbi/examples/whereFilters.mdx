# whereFilters

## bar-by-province-with-filter

柱状图 - 按省份销售额（带过滤条件）

```tsx preview
import { VBI } from '@visactor/vbi'
import { VSeedRender } from '@components'
import { useEffect, useState } from 'react'

export default () => {
  const [vseed, setVSeed] = useState(null)

  useEffect(() => {
    const run = async () => {
      const schema = [
        { name: 'province', type: 'string' },
        { name: 'sales', type: 'number' },
      ]
      const dataset = [
        { province: 'province0', sales: 100 },
        { province: 'province1', sales: 200 },
        { province: 'province2', sales: 300 },
        { province: 'province3', sales: 400 },
        { province: 'province4', sales: 500 },
      ]

      VBI.registerConnector('demo', async () => ({
        discoverSchema: async () => schema,
        query: async () => ({ dataset }),
      }))

      const builder = VBI.from({
        connectorId: 'demo',
        chartType: 'bar',
        dimensions: [{ field: 'province', alias: '省份' }],
        measures: [{ field: 'sales', alias: '销售额', encoding: 'xAxis', aggregate: { func: 'sum' } }],
        whereFilters: [],
        theme: 'light',
        locale: 'zh-CN',
        version: 1,
      })

      const applyBuilder = (builder: VBIBuilder) => {
        builder.whereFilters.add('sales', (node) => {
          node.setOperator('gt').setValue(100000)
        })
      }
      applyBuilder(builder)

      const result = await builder.buildVSeed()
      setVSeed(result)
    }
    run()
  }, [])

  if (!vseed) return <div>Loading...</div>

  return <VSeedRender vseed={vseed} />
}
```

## column-by-area-multiple-filters

柱状图 - 按区域销售额（多过滤条件）

```tsx preview
import { VBI } from '@visactor/vbi'
import { VSeedRender } from '@components'
import { useEffect, useState } from 'react'

export default () => {
  const [vseed, setVSeed] = useState(null)

  useEffect(() => {
    const run = async () => {
      const schema = [
        { name: 'area', type: 'string' },
        { name: 'sales', type: 'number' },
      ]
      const dataset = [
        { area: 'area0', sales: 100 },
        { area: 'area1', sales: 200 },
        { area: 'area2', sales: 300 },
        { area: 'area3', sales: 400 },
        { area: 'area4', sales: 500 },
      ]

      VBI.registerConnector('demo', async () => ({
        discoverSchema: async () => schema,
        query: async () => ({ dataset }),
      }))

      const builder = VBI.from({
        connectorId: 'demo',
        chartType: 'column',
        dimensions: [{ field: 'area', alias: '区域' }],
        measures: [{ field: 'sales', alias: '销售额', encoding: 'yAxis', aggregate: { func: 'sum' } }],
        whereFilters: [],
        theme: 'light',
        locale: 'zh-CN',
        version: 1,
      })

      const applyBuilder = (builder: VBIBuilder) => {
        builder.whereFilters
          .add('sales', (node) => node.setOperator('gt').setValue(500000))
          .add('area', (node) => node.setOperator('in').setValue(['华东', '华北']))
      }
      applyBuilder(builder)

      const result = await builder.buildVSeed()
      setVSeed(result)
    }
    run()
  }, [])

  if (!vseed) return <div>Loading...</div>

  return <VSeedRender vseed={vseed} />
}
```
