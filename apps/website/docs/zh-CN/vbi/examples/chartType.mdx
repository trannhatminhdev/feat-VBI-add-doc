# chartType

## area-by-order-date

面积图 - 按日期销售额趋势

```tsx preview
import { VBI } from '@visactor/vbi'
import { VSeedRender } from '@components'
import { useEffect, useState } from 'react'

export default () => {
  const [vseed, setVSeed] = useState(null)

  useEffect(() => {
    const run = async () => {
      const schema = [
        { name: 'order_date', type: 'string' },
        { name: 'sales', type: 'number' },
      ]
      const dataset = [
        { order_date: 'order_date0', sales: 100 },
        { order_date: 'order_date1', sales: 200 },
        { order_date: 'order_date2', sales: 300 },
        { order_date: 'order_date3', sales: 400 },
        { order_date: 'order_date4', sales: 500 },
      ]

      VBI.registerConnector('demo', async () => ({
        discoverSchema: async () => schema,
        query: async () => ({ dataset }),
      }))

      const builder = VBI.from({
        connectorId: 'demo',
        chartType: 'area',
        dimensions: [{ field: 'order_date', alias: '订单日期' }],
        measures: [{ field: 'sales', alias: '销售额', encoding: 'yAxis', aggregate: { func: 'sum' } }],
        filters: [],
        theme: 'light',
        locale: 'zh-CN',
        version: 1,
      })

      const applyBuilder = (builder: VBIBuilder) => {
        builder.chartType.changeChartType('line')
      }
      applyBuilder(builder)

      const result = await builder.buildVSeed()
      setVSeed(result)
    }
    run()
  }, [])

  if (!vseed) return <div>Loading...</div>

  return <VSeedRender vseed={vseed} />
}
```
