# dimensions

## add-dimension

添加维度

```tsx preview
import { VBI } from '@visactor/vbi'
import { VSeedRender } from '@components'
import { useEffect, useState } from 'react'

export default () => {
  const [vseed, setVSeed] = useState(null)

  useEffect(() => {
    const run = async () => {
      const schema = [
        { name: 'product_type', type: 'string' },
        { name: 'sales', type: 'number' },
      ]
      const dataset = [
        { product_type: 'product_type0', sales: 100 },
        { product_type: 'product_type1', sales: 200 },
        { product_type: 'product_type2', sales: 300 },
        { product_type: 'product_type3', sales: 400 },
        { product_type: 'product_type4', sales: 500 },
      ]

      VBI.registerConnector('demo', async () => ({
        discoverSchema: async () => schema,
        query: async () => ({ dataset }),
      }))

      const builder = VBI.from({
        connectorId: 'demo',
        chartType: 'table',
        dimensions: [],
        measures: [],
        whereFilters: [],
        havingFilters: [],
        theme: 'light',
        locale: 'zh-CN',
        version: 1,
        limit: 20,
      })

      const applyBuilder = (builder: VBIBuilder) => {
        builder.dimensions.add('product_type', (node) => {
          node.setAlias('产品类型')
        })
      }
      applyBuilder(builder)

      const result = await builder.buildVSeed()
      setVSeed(result)
    }
    run()
  }, [])

  if (!vseed) return <div>Loading...</div>

  return <VSeedRender vseed={vseed} />
}
```
